                placeholder="Type your word or sentence here (e.g., 'hello world')"

                oninput="translateText()"

            ></textarea>

        </section>

            <div id="boirine-output"

                 class="w-full min-h-32 p-4 text-xl font-mono text-green-800 bg-green-50 border-2 border-green-300 rounded-lg select-all"

            >

                <!-- Translation will appear here -->

                <span class="text-gray-400 italic text-base">Translation will appear here instantly...</span>

            </div>

            <button onclick="copyToClipboard()"

                    class="mt-4 w-full md:w-auto px-6 py-2 bg-indigo-600 text-white font-medium rounded-lg shadow-md hover:bg-indigo-700 transition duration-150 focus:outline-none focus:ring-2 focus:ring-indigo-500 focus:ring-offset-2 disabled:opacity-50"

                    id="copy-button" disabled

            >

                Copy Boirine Text

            </button>

                Text copied to clipboard!

            </p>

        <!-- Key/Legend (Optional but helpful) -->

        <details class="mt-8 border-t pt-4">

            <summary class="text-gray-600 font-medium cursor-pointer">View Boirine Alphabet Key</summary>

            <div id="key-legend" class="mt-4 grid grid-cols-4 sm:grid-cols-6 gap-2 text-sm text-gray-600">

                <!-- Key generated by JS for easy reference -->

            </div>

    <script>

        // The core mapping for the Boirine Language

        const BOIRINE_MAP = {

            'A': 'Chi', 'B': 'Si', 'C': 'Uo', 'D': 'Fo', 'E': 'Li',

            'F': 'Lou', 'G': 'Cgha', 'H': 'Bi', 'I': 'Lia', 'J': 'Sia',

            'K': 'Hk', 'L': 'La', 'M': 'Sea', 'N': 'Cla', 'O': 'Lo',

            'P': 'Pa', 'Q': 'Qu', 'R': 'Ri', 'S': 'Mi', 'T': 'Ti',

            'U': 'Ou', 'V': 'Vo', 'W': 'Wo', 'X': 'Xio', 'Y': 'Yua',

            'Z': 'Zuo'

        };


        const englishInput = document.getElementById('english-input');

        const boirineOutput = document.getElementById('boirine-output');

        const copyButton = document.getElementById('copy-button');

        const copyMessage = document.getElementById('copy-message');


        /**

         * Translates the English text in the input area to Boirine.

         */

        function translateText() {

            const input = englishInput.value;

            let boirineText = '';


            // Convert input to uppercase for lookup consistency

            const upperInput = input.toUpperCase();


            for (let i = 0; i < upperInput.length; i++) {

                const char = upperInput[i];

                const translation = BOIRINE_MAP[char];


                if (translation) {

                    // Append the Boirine syllable if found in the map

                    boirineText += translation;

                } else {

                    // Keep non-alphabetic characters (spaces, numbers, punctuation) as is

                    boirineText += input[i]; 

                }

            }


            // Update the output display

            if (boirineText.trim() === '') {

                boirineOutput.innerHTML = '<span class="text-gray-400 italic text-base">Translation will appear here instantly...</span>';

                copyButton.disabled = true;

            } else {

                boirineOutput.textContent = boirineText;

                copyButton.disabled = false;

            }

        }


        /**

         * Copies the generated Boirine text to the clipboard.

         */

        function copyToClipboard() {

            const textToCopy = boirineOutput.textContent;

            

            // Temporary element for copy

            const tempInput = document.createElement('textarea');

            tempInput.value = textToCopy;

            document.body.appendChild(tempInput);

            tempInput.select();

            document.execCommand('copy');

            document.body.removeChild(tempInput);


            // Show confirmation message

            copyMessage.classList.remove('opacity-0');

            copyMessage.classList.add('opacity-100');

            setTimeout(() => {

                copyMessage.classList.remove('opacity-100');

                copyMessage.classList.add('opacity-0');

            }, 2000);

        }


        /**

         * Generates and displays the alphabet key for reference.

         */

        function generateKeyLegend() {

            const keyLegend = document.getElementById('key-legend');

            let html = '';

            for (const [english, boirine] of Object.entries(BOIRINE_MAP)) {

                html += `<div class="p-1 rounded-md bg-gray-100 text-center"><span class="font-bold text-indigo-500">${english}</span> = ${boirine}</div>`;

            }

            keyLegend.innerHTML = html;

        }


        // Run on load

        window.onload = generateKeyLegend;


    </script>
